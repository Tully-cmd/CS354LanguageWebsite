            class Bank {
            constructor() {
                this.accounts = new Set();
            }
            
            add(account) {
                this.accounts.add(account);
            }
            
            accrue(rate) {
                for (let account of this.accounts) {
                    account.accrue(rate);
                }
            }
            
            toString() {
                let result = "";
                for (let account of this.accounts) {
                    result += account + "\n";
                }
                return result;
            }
